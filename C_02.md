### 리스트(List)자료형
### 리스트명 = [요소1, 요소2, 요소3,...]
```python
a = [1,2,3]
b = [4,5,6]
a[0] + b[2]

# 정렬
a.sort()

#순서 뒤집기
a.reverse()

# 항목 추가
a.append(7)

# 지정된 위치에 값 추가
a.insert(0,9) 

# 리스트에서 첫 번째로 나오는 x 삭제
a.remove(7)

# 마지막 요소 돌려주고 삭제
a.pop()

# 수정과 삭제
a = [1,2,3]
a[2] = 4

del a[1

# 요소 x의 개수 세기
a = [1,2,3,1]
a.count(1)

# 리스트 확장
a = [1,2,3]
a.extend([4,5])
a

b = [6,7]
a.extend(b)
a

```

### 딕셔너리(Dictionary)자료형
```python
dic = {'name':'nam', 'phone':'010-111-1111', 'address':'원주'}
dic

# 데이터 추가
dic['birth'] = '2022-02-01'
dic

# 키값 얻기
dic['name']

# 삭제
del dic['name']
dic

# 키값 확인
dic.keys()

# value  확인하기
dic.items()

# value 얻기
dic.get('address')

```

### 집합 자료형(Set)

```python
s1 = set([1,2,3])
s1

s2 = set('Hello')
s2

#인덱싱 접근방법
s1 = set([1,2,3])
l1 = list(s1)
l1[2]

t1 = tuple(s1)
t1[0]

# 교집합
s1 = set([1,2,3,4,5,6])
s2 = set([4,5,6,7,8,9])

s1.intersection(s2)
# s1 & s2

# 합집합
s1.union(s2)
s1 | s2

# 차집합
s1.difference(s2)
s1 - s2

#값 추가
s1 = set([1,2,3])
s1.add(4)
s1

# 여러개 추가
s1.update([4,5,6])
s1

# 삭제
s1.remove(2)
s1

```
